abstract class HashTable<T> {

  // Конструктор
  // постусловие: создана пустая хэш-таблица с указанным максимальным размером
  public HashTable<T> HashTable(int maxSize);

  // Команды

  // постусловие: добавлен новый элемент value в хэш-таблицу
  // предусловие: таблица не переполнена
  public void add(T value);

  // постусловие: элемент value удалён из хэш-таблицы, если он присутствовал
  public void remove(T value);

  // постусловие: все элементы удалены из хэш-таблицы
  public void clear();

  // Запросы

  // постусловие: возвращает true, если элемент value присутствует в хэш-таблице
  public boolean contains(T value);

  // постусловие: возвращает количество элементов в хэш-таблице
  public int size();

  // постусловие: возвращает максимальный размер хэш-таблицы
  public int capacity();

  // Запросы статусов
  public int get_add_status();    // успешно; таблица переполнена; элемент уже существует
  public int get_remove_status(); // успешно; элемент не найден
  public int get_contains_status(); // успешно; элемент не найден
}
