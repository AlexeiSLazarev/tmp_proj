abstract class NativeDictionary<V> {

  // Конструктор
  // постусловие: создан пустой словарь с указанным максимальным размером
  public NativeDictionary<V> NativeDictionary(int maxSize);

  // Команды

  // постусловие: добавлена новая пара (ключ, значение) в словарь
  // предусловие: ключ уникален или перезаписывается
  public void put(String key, V value);

  // постусловие: элемент с указанным ключом удалён из словаря
  public void remove(String key);

  // постусловие: все элементы удалены из словаря
  public void clear();

  // Запросы

  // постусловие: возвращает значение, связанное с ключом, если он присутствует в словаре
  public V get(String key);

  // постусловие: возвращает true, если ключ присутствует в словаре
  public boolean contains(String key);

  // постусловие: возвращает количество пар ключ-значение в словаре
  public int size();

  // постусловие: возвращает максимальный размер словаря
  public int capacity();

  // Запросы статусов
  public int get_put_status();     // успешно; словарь переполнен; ключ уже существует
  public int get_remove_status();  // успешно; ключ не найден
  public int get_get_status();     // успешно; ключ не найден
  public int get_contains_status();// успешно; ключ не найден
}
